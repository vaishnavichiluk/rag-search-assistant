
# Streamlit UI Framework-- for building user interface
pip install streamlit              

# LangChain Ecosystem
pip install langchain              # Core LangChain
pip install langchain-community    # Vector stores, loaders, embeddings
pip install langchain-core         # LangChain runnable pipeline support

# Embeddings & Vector Database

pip install sentence-transformers  # To create embeddings for document chunks
pip install faiss-cpu              # FAISS vector database for similarity search


# HuggingFace Models (FLAN-T5)

pip install transformers           # To load and run HuggingFace LLMs
pip install accelerate             # For optimized CPU inference
pip install sentencepiece          # Tokenizer support for T5-style models


# PDF Processing

pip install pypdf                  # For reading PDF files into text
# OR
pip install pymupdf                # More advanced PDF extraction (optional)
